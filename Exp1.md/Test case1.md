def xor_gate(x1, x2):
    return x1 ^ x2  # XOR operation

    [0, 1],
    [0, 1],
    [0, 0],
    [1, 1]
]

print("XOR Problem Test Case Results:")

    output = xor_gate(x[0], x[1])
